package myPackage;

import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;
import java.util.List;

public class Program {

	public static void main(String[] args) {

		List<Person> list = new ArrayList<>();
		list.add(new Person("taro", 10));
		list.add(new Person("jiro", 15));
		list.add(new Person("jiro", 20));

		// ソート例１  ※名前の昇順、年齢の降順
		list.sort(Comparator.comparing(Person::getName, Comparator.naturalOrder())
				            .thenComparing(Person::getAge, Comparator.reverseOrder()));

		System.out.println(list);

		// ソート例２  ※名前の降順、年齢の昇順
		Collections.sort(list, new Comparator<Person>() {
			@Override
			public int compare(Person x, Person y) {
				// Nameの降順、年齢の昇順とする
				int compName = x.getName().compareTo(y.getName()) * -1;
				int compAge = Integer.compare(x.getAge(), y.getAge());
				return compName != 0 ? compName : compAge;
			}
		});

		System.out.println(list);
	}
}

// 独自クラス
class Person {
	private String name = "";
	private int age = 0;

	public Person(String name, int age) {
		this.name = name;
		this.age = age;
	}

	public String getName() { return this.name; }
	public int getAge() { return this.age; }

	@Override
	public String toString() {
		return "name: " + getName() + " age: " + getAge() + "\n";
	}

}
